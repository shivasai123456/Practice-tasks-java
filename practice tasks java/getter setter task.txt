public class App {
    private String name;
 
    // Getter
    public String getName() {
      return name;
    }
 
    // Setter
    public void setName(String newName) {
      this.name = newName;
    }

    public static void main(String[] args) {
        App myObj = new App();

        System.out.println(myObj.name);

    }
